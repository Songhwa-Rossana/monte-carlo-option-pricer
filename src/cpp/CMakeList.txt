cmake_minimum_required(VERSION 3.15)
project(MonteCarloPricer VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Compiler flags for optimization
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    set(CMAKE_CXX_FLAGS_RELEASE "-O3 -march=native -DNDEBUG")
endif()

# Find OpenMP for multi-threading
find_package(OpenMP)
if(OpenMP_CXX_FOUND)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
endif()

# Include directories
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

# Source files
set(SOURCES
    src/options.cpp
    src/monte_carlo.cpp
    src/random_generator.cpp
    src/variance_reduction.cpp
    src/greeks.cpp
)

# Create static library
add_library(monte_carlo_lib STATIC ${SOURCES})

if(OpenMP_CXX_FOUND)
    target_link_libraries(monte_carlo_lib PUBLIC OpenMP::OpenMP_CXX)
endif()

# Optional: Build Python bindings with pybind11
option(BUILD_PYTHON_BINDINGS "Build Python bindings" ON)

if(BUILD_PYTHON_BINDINGS)
    find_package(pybind11 REQUIRED)
    pybind11_add_module(monte_carlo_cpp src/python_bindings.cpp)
    target_link_libraries(monte_carlo_cpp PRIVATE monte_carlo_lib)
endif()

# Optional: Build tests
option(BUILD_TESTS "Build tests" ON)
if(BUILD_TESTS)
    enable_testing()
    add_subdirectory(tests)
endif()
